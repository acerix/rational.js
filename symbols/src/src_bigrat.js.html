191 /**
192  * Adds two bigrats
193  *
194  * @param {bigrat} out the receiving number
195  * @param {bigrat} a the first operand
196  * @param {bigrat} b the second operand
197  * @returns {bigrat} out
198  */
199 bigrat.add = function(out, a, b) {
200 	if (a[1].compare(b[1])===0) {
201 		out[0] = a[0].add(b[0]);
202 		out[1] = a[1];
203 	}
204 	else {
205 		out[0] = a[0].multiply(b[1]).add( b[0].multiply(a[1]) );
206 		out[1] = a[1].multiply(b[1]);
207 	}
208 	return bigrat.normalize(out, out);
209 };

291 /**
292  * Returns true when the first bigrat is equal to the second
293  *
294  * @param {bigrat} a the first operand
295  * @param {bigrat} b the second operand
296  * @returns {Bool} true when the two bigrats are equal
297  */
298 bigrat.equals = function(a, b) {
299 	if (a[0].isZero() && b[0].isZero()) return true; // both are Zero
300 	if (a[1].isZero() && b[1].isZero()) return true; // both are Infinity
301 	return (a[0].compare(b[0])===0) && (a[1].compare(b[1])===0);
302 };

304 /**
305  * Returns true when the first bigrat is approximately equal to the second
306  *
307  * @param {bigrat} a the first operand
308  * @param {bigrat} b the second operand
309  * @returns {Bool} true when the difference between the two bigrats is less than bigrat.INFINITESIMAL
310  */
311 bigrat.approximates = function(a, b) {
312 	if (bigrat.equals(a, b)) return true;
313     var d = bigrat.create();
314     bigrat.sub(d, a, b);
315     bigrat.abs(d, d);
316     return bigrat.isLessThan(d, bigrat.INFINITESIMAL);
317 };

422 /**
423  * Normalize a bigrat
424  *
425  * @param {bigrat} out the receiving number
426  * @param {bigrat} a number to normalize
427  * @returns {bigrat} out
428  */
429 bigrat.normalize = function(out, a) {
430 	//if (isNaN(a[0])||isNaN(a[1])) return out = bigrat.clone(bigrat.INFINULL);
431 	if (a[0].isZero()||a[1].isZero()) return out = a;
432 	if (a[0].compare(a[1])===0) return out = bigrat.clone(bigrat.ONE);
433 	if (!a[1].isNegative()) {
434 		out[0] = a[0];
435 		out[1] = a[1];
436 		if (out[1].isZero()) return out
437 	}
438 	else {
439 		out[0] = a[0].negate();
440 		out[1] = a[1].negate();
441 	}
442 	var gcd = bigint.greatest_common_divisor(out[0].abs(), out[1]);
443 	if (gcd.compare(BigInteger.ONE)>0) {
444 		out[0] = out[0].quotient(gcd);
445 		out[1] = out[1].quotient(gcd);
446 	}
447 	return out;
448 };

475 /**
476  * Raises a bigrat to an integer exponent
477  *
478  * @param {bigrat} out the receiving number
479  * @param {bigrat} a the number to exponentiate
480  * @param {Integer} p power to raise the number by
481  * @returns {bigrat} out
482  */
483 bigrat.power = function(out, a, p) {
484 	if (p===2) {
485 		out[0] = a[0].square();
486 		out[1] = a[1].square();
487 	}
488 	else if (p>0) {
489 		out[0] = a[0].pow(p);
490 		out[1] = a[1].pow(p);
491 	}
492 	else if (p<0) {
493 		p = Math.abs(p);
494 		out[0] = a[1].pow(p);
495 		out[1] = a[0].pow(p);
496 	}
497 	else {
498 		bigrat.copy(out, bigrat.ONE);
499 	}
500 	return out;
501 };

520 /**
521  * Find a bigrat approximation which equals the input bigrat when raised to the given integer exponent
522  * 
523  * Newton's method converges alot faster... could that be used to find the pattern in the SB tree?
524  *
525  * @param {bigrat} out the receiving number
526  * @param {bigrat} a the number to find the root of
527  * @param {Integer} n
528  * @returns {bigrat} out
529  */
530 bigrat.nthRoot = function (out, a, n) {
531 	if (bigrat.equals(a, bigrat.ZERO)) return bigrat.copy(out, bigrat.ZERO);
532 	if (bigrat.equals(a, bigrat.ONE)) return bigrat.copy(out, bigrat.ONE);
533 	if (bigrat.equals(a, bigrat.INFINITY)) return bigrat.copy(out, bigrat.INFINITY);
534 	if (bigrat.equals(a, bigrat.INFINULL)) return bigrat.copy(out, bigrat.INFINULL);
535 	
536 	var neg = bigrat.isNegative(a);
537 	if (neg) a[0] = a[0].negate();
538 	
539 	bigrat.copy(out, bigrat.ONE);
540 	var m = [
541 		BigInteger(1),
542 		BigInteger(0),
543 		BigInteger(0),
544 		BigInteger(1)
545 	];
546 	var test = bigrat.clone(bigrat.ONE);
547 	
548 	var c = BIGRAT_MAX_LOOPS;
549 	while ( !bigrat.approximates(a, test) && c-- ) {
550 		if (bigrat.isLessThan(a, test)) {
551 			m[0] = m[0].add(m[1]);
552 			m[2] = m[2].add(m[3]);
553 		}
554 		else {
555 			m[1] = m[1].add(m[0]);
556 			m[3] = m[3].add(m[2]);
557 		}
558 		out[0] = m[0].add(m[1]);
559 		out[1] = m[2].add(m[3]);
560 		bigrat.pow(test, out, n);
561 	}
562 	
563 	if (neg) { 
564 		a[0] = a[0].negate();
565 		if (n%2===1) bigrat.neg(out, out);
566 	}
567 	
568 	return out;
569 };

722 	a = parseFloat(a);
723 	if (a===0) return bigrat.copy(out, bigrat.ZERO);
724 	if (a===1) return bigrat.copy(out, bigrat.ONE);
725 	if (a===Infinity) return bigrat.copy(out, bigrat.INFINITY);
726 	if (isNaN(a)) return bigrat.copy(out, bigrat.INFINULL);
727 	if (a%1===0) return bigrat.fromInteger_copy(out, a);
728 	if ((1/a)%1===0) return bigrat.fromIntegerInverse_copy(out, parseInt(1/a));
729 	
730 	bigrat.copy(out, bigrat.ONE);
731 	
732 	var m = [
733 		BigInteger(1),
734 		BigInteger(0),
735 		BigInteger(0),
736 		BigInteger(1)
737 	];
738 	var test = a;
739 	
740 	// traverse the Stern-Brocot tree until a match is found
741 	var c = BIGRAT_MAX_LOOPS;
742 	while ( Math.abs(out[0].valueOf() - test) > bigrat.EPSILON && c-- ) {
743 		if (out[0].valueOf() > test) {
744 			m[0] = m[0].add(m[1]);
745 			m[2] = m[2].add(m[3]);
746 		}
747 		else {
748 			m[1] = m[1].add(m[0]);
749 			m[3] = m[3].add(m[2]);
750 		}
751 		out[0] = m[0].add(m[1]);
752 		out[1] = m[2].add(m[3]);
753 		
754 		test = out[1].valueOf() * a;
755 	}
756 	return out;
757 };

771 /**
772  * Parametric sine: 2a / (1 + a²)
773  *
774  * @param {bigrat} out the receiving number
775  * @param {bigrat} a number for which to calculate the parametric sine
776  * @returns {bigrat} out
777  */
778 bigrat.sin = function(out, a) {
779 	if (a[1].isZero()) return bigrat.copy(out, bigrat.ZERO);
780 	bigrat.scalar_multiply(out, a, 2);
781 	var d = bigrat.create();
782 	bigrat.pow(d, a, 2);
783 	bigrat.add(d, d, bigrat.ONE);
784 	bigrat.divide(out, out, d);
785 	return out;
786 };

788 /**
789  * Parametric cosine: (1 - a²) / (1 + a²)
790  *
791  * @param {bigrat} out the receiving number
792  * @param {bigrat} a number for which to calculate the parametric cosine
793  * @returns {bigrat} out
794  */
795 bigrat.cos = function(out, a) {
796 	if (a[1].isZero()) return bigrat.neg(out, bigrat.ONE);
797 	var a2 = bigrat.create();
798 	bigrat.pow(a2, a, 2);
799 	bigrat.sub(out, bigrat.ONE, a2);
800 	var d = bigrat.create();
801 	bigrat.add(d, bigrat.ONE, a2);
802 	bigrat.divide(out, out, d);
803 	return out;
804 };

806 /**
807  * Parametric tangent: sin(a) / cos(a)
808  *
809  * @param {bigrat} out the receiving number
810  * @param {bigrat} a number for which to calculate the parametric tangent
811  * @returns {bigrat} out
812  */
813 bigrat.tan = function(out, a) {
814 	bigrat.scalar_multiply(out, a, 2); // out = a * 2
815 	var t = bigrat.create();
816 	bigrat.pow(t, a, 2); // t = a * a
817 	bigrat.scalar_multiply(t, t, 2); // t *= 2
818 	bigrat.add(out, out, t); // out += t
819 	bigrat.pow(t, a, 4); // t = a * a * a * a
820 	bigrat.sub(t, bigrat.ONE, t); // t = 1 - t
821 	bigrat.divide(out, out, t);
822 	return out;
823 };

825 /**
826  * Returns an Egyptian representation
827  * The returned string can be evaluated in "calc - arbitrary precision calculator"
828  *
829  * @param {bigrat} a number to represent as an Egyptian fraction
830  * @returns {String} string representing the most simple sum of fractions having a numebigrator of one, in "calc" format
831  */
832 bigrat.toEgyptian = function (a) {
833 	var t = bigrat.clone(a);
834 	bigrat.abs(t, t);
835 	var b = bigrat.floor(t);
836 	if (b) bigrat.sub(t, t, bigrat.fromInteger(b));
837 	if (!t[0]) return b.toString();
838 	if (!b) b = '';
839 	var d = 1;
840 	var f = bigrat.create();
841 	while (t[0] !== 1) {
842 		d++;
843 		f = bigrat.fromValues(1, d);
844 		if (bigrat.isGreaterThan(t, f)) {
845 			if (b) b += ' + ';
846 			b += bigrat.str(f);
847 			bigrat.sub(t, t, f);
848 		}
849 	}
850 	if (!t) {
851 		if (!b) return '0';
852 		return b;
853 	}
854 	return b ? b + ' + ' + bigrat.str(t) : bigrat.str(t);
855 };

857 /**
858  * Returns a Babylonian representation (base 60)
859  * The returned string can be evaluated in "calc - arbitrary precision calculator"
860  *
861  * @param {bigrat} a number to represent as a Babylonian fraction
862  * @returns {String} string containing the decimal representations of the base 60 digits and their powers, in "calc" format
863  */
864 bigrat.toBabylonian = function (a) {
865 	var s = '';
866 	 // there must be a better way to do this, avoiding conversion to decimal
867 	var t = bigrat.toDecimal(a);
868 	var n = parseInt(t);
869 	var r = t - n;
870 	var d = 0;
871 	var p = 0;
872 	while (n > 0) {
873 		d = n % 60;
874 		if (d) s = d + ' * 60^' + p + ( s ? ' + ' : '' ) + s;
875 		n = (n - d) / 60;
876 		p++;
877 	}
878 	p = -1;
879 	while (r > 0) {
880 		r *= 60;
881 		d = parseInt(r + 1E-13);
882 		r = r - d;
883 		if (r < -1E-13) continue;
884 		if (d) s += ( s ? ' + ' : '' ) + d + ' * 60^' + p;
885 		p--;
886 	}
887 	return s ? s : '0';
888 };

890 /**
891  * Returns a string of L's and R's representing the Stern Brocot path
892  *
893  * @param {bigrat} a number to trace in the Stern Brocot tree
894  * @returns {String} Stern Brocot path
895  */
896 bigrat.traceSternBrocot = function (a) {
897 	var path = '';
898 	if (
899 		bigrat.equals(a, bigrat.ZERO)
900 	||
901 		bigrat.equals(a, bigrat.ONE)
902 	||
903 		bigrat.equals(a, bigrat.INFINITY)
904 	||
905 		bigrat.equals(a, bigrat.INFINULL)
906 		) return path;
907 
908 	var neg = bigrat.isNegative(a);
909 	if (neg) a[0] = a[0].negate();
910 	
911 	var r = bigrat.clone(bigrat.ONE);
912 	
913 	var m = [
914 		BigInteger(1),
915 		BigInteger(0),
916 		BigInteger(0),
917 		BigInteger(1)
918 	];
919 	
920 	var r_streak = 0;
921 	var l_streak = 0;
922 	
923 	var c = 65536;
924 	while ( !bigrat.equals(a, r) && c-- ) {
925 		if (bigrat.isLessThan(a, r)) {
926 			m[0] = m[0].add(m[1]);
927 			m[2] = m[2].add(m[3]);
928 			l_streak++;
929 			if (r_streak) {
930 				path += 'R';
931 				if (r_streak!==1) path += r_streak;
932 				r_streak = 0;
933 				path += ' ';
934 			}
935 		}
936 		else {
937 			m[1] = m[1].add(m[0]);
938 			m[3] = m[3].add(m[2]);
939 			r_streak++;
940 			if (l_streak) {
941 				path += 'L';
942 				if (l_streak!==1) path += l_streak;
943 				l_streak = 0;
944 				path += ' ';
945 			}
946 		}
947 		r[0] = m[0].add(m[1]);
948 		r[1] = m[2].add(m[3]);
949 	}
950 	if (l_streak) {
951 		path += 'L';
952 		if (l_streak!==1) path += l_streak;
953 	}
954 	else if (r_streak) {
955 		path += 'R';
956 		if (r_streak!==1) path += r_streak;
957 	}
958 	
959 	if (c<0) path += '...';
960 	
961 	if (neg) a[0] = a[0].negate();
962 	
963 	return path;
964 };

966 /**
967  * Returns an array of integers representing the continued fraction
968  *
969  * @param {bigrat} a number to convert to a continued fraction
970  * @param {Integer} maximum number of iterations
971  * @returns {Array} integers of the continued fraction
972  */
973 bigrat.toContinuedFraction = function (a, loop_limit) {
974 	loop_limit = typeof loop_limit==='undefined' ? 65536 : parseInt(loop_limit);
975 	if (bigrat.equals(a, bigrat.ZERO)) return [0];
976 	if (bigrat.equals(a, bigrat.ONE)) return [1];
977 	if (bigrat.equals(a, bigrat.NEGONE)) return [-1];
978 	if (bigrat.equals(a, bigrat.INFINITY)) return [1, 0];
979 	if (bigrat.equals(a, bigrat.INFINULL)) return [0, 0];
980 
981 	var neg = bigrat.isNegative(a);
982 	if (neg) a[0] = a[0].negate();
983 	
984 	var r = bigrat.clone(bigrat.ONE);
985 	
986 	var m = [
987 		BigInteger(1),
988 		BigInteger(0),
989 		BigInteger(0),
990 		BigInteger(1)
991 	];
992 	
993 	var direction = 1;
994 	var result = [0];
995 	var result_last = result.length - 1;
996 	
997 	while ( !bigrat.equals(a, r) && loop_limit-- ) {
998 		if (bigrat.isLessThan(a, r)) {
999 			if (direction===-1) {
1000 				result[result_last]++;
1001 			}
1002 			else {
1003 				direction = -1;
1004 				result.push(1);
1005 				result_last++;
1006 			}
1007 			m[0] = m[0].add(m[1]);
1008 			m[2] = m[2].add(m[3]);
1009 		}
1010 		else {
1011 			if (direction===1) {
1012 				result[result_last]++;
1013 			}
1014 			else {
1015 				direction = 1;
1016 				result.push(1);
1017 				result_last++;
1018 			}
1019 			m[1] = m[1].add(m[0]);
1020 			m[3] = m[3].add(m[2]);
1021 		}
1022 		r[0] = m[0].add(m[1]);
1023 		r[1] = m[2].add(m[3]);
1024 	}
1025 
1026 	// add a zero to the end to indicate an incomplete result
1027 	if (loop_limit<0) result.push(0);
1028 	else result[result_last]++;
1029 	
1030 	if (neg) for (var i in result) result[i] = -result[i];
1031 	
1032 	return result;
1033 };

1035 /**
1036  * Returns a bigrat from an array of integers representing a continued fraction
1037  *
1038  * @param {bigrat} out the receiving number
1039  * @param {Array} integers of the continued fraction
1040  * @returns {bigrat} out
1041  */
1042 bigrat.fromContinuedFraction = function(out, cf) {
1043 	bigrat.fromInteger_copy(out, cf[cf.length-1]);
1044 	for (var i=cf.length-2;i>-1;i--) {
1045 		bigrat.invert(out, out);
1046 		bigrat.add(out, bigrat.fromInteger(cf[i]), out);
1047 	}
1048 	return out;
1049 };
