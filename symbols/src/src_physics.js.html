 25 /**
 26  * @class Physics
 27  * @name physics
 28  * @requires bigrat
 29  */
 30 var physics = {};

 32 /**
 33  * Fine-structure constant (α), according to CODATA 2011
 34  *
 35  * @property FINE_STRUCTURE
 36  * @type bigrat
 37  * @static
 38  * @final
 39  */
 40 physics.FINE_STRUCTURE = bigrat.fromValues(30223, 4141639);
 41 
 42 /**
 43  * Alias for {@link physics.FINE_STRUCTURE}
 44  * @type bigrat
 45  * @static
 46  * @final
 47  */
 48 physics.a = physics.FINE_STRUCTURE;
 49 
 50 /**
 51  * Approximately the fine-structure constant, generated mathematically
 52  *
 53  * @property FINE_STRUCTURE_ALT
 54  * @type bigrat
 55  * @static
 56  * @final
 57  */
 58 physics.FINE_STRUCTURE_ALT = bigrat.fromValues(13355891, 1830237867);

 60 /**
 61  * Planck's constant (h) in Joule seconds
 62  *
 63  * @property PLANK
 64  * @type bigrat
 65  * @static
 66  * @final
 67  */
 68 physics.PLANK = bigrat.fromValues(662606957, 1e42);
 69 
 70 /**
 71  * Alias for {@link physics.PLANKS}
 72  * @type bigrat
 73  * @static
 74  * @final
 75  */
 76 physics.h = physics.PLANK;

 78 /**
 79  * Reduced Planck's constant (h / 2π)

 81  * @property HBAR

 83  * @static
 84  * @final
 85  */

 88 /**
 89  * Gravitational constant (G) in m³ / kg s²

 91  * @property GRAVITATIONAL

 94  * @final
 95  */

 98 /**
 99  * Alias for {@link physics.GRAVITATIONAL}

101  * @static
102  * @final
103  */
104 physics.G = physics.GRAVITATIONAL;

106 /**
107  * The speed of light in vacuum (c) in m / s

109  * @property SPEED_OF_LIGHT
