 21 /**
 22  * @class Physics
 23  * @name physics
 24  * @requires bigrat
 25  */
 26 var physics = {};

 28 /**
 29  * Fine-structure constant (α), according to CODATA 2011
 30  *
 31  * @property FINESTRUCTURE
 32  * @type bigrat
 33  * @static
 34  * @final
 35  */
 36 physics.FINESTRUCTURE = bigrat.fromValues(30223, 4141639);
 37 
 38 /**
 39  * Alias for {@link physics.FINESTRUCTURE}
 40  * @type bigrat
 41  * @static
 42  * @final
 43  */
 44 physics.a = physics.FINESTRUCTURE;
 45 
 46 /**
 47  * Approximately the fine-structure constant, generated mathematically from a limit
 48  *
 49  * @property FINESTRUCTURE_ALT
 50  * @type bigrat
 51  * @static
 52  * @final
 53  */
 54 physics.FINESTRUCTURE_ALT = bigrat.fromValues(13355891, 1830237867);
 55 
 56 /**
 57  * Planck's constant (h) in Joule seconds
 58  *
 59  * @property PLANK
 60  * @type bigrat
 61  * @static
 62  * @final
 63  */
 64 physics.PLANK = bigrat.fromValues(662606957, 1e42);
 65 
 66 /**
 67  * Alias for {@link physics.PLANKS}
 68  * @type bigrat
 69  * @static
 70  * @final
 71  */
 72 physics.h = physics.PLANK;
 73 
 74 /**
 75  * Reduced Planck's constant (h / 2π)
 76  *
 77  * @property HBAR
 78  * @type bigrat
 79  * @static
 80  * @final
 81  */
 82 physics.hBAR = bigrat.fromValues(1054571726, 1e43);

 84 /**
 85  * Gravitational constant (G) in m³ / kg s²

 87  * @property GRAVITATIONAL

 89  * @static
 90  * @final
 91  */

 94 /**
 95  * Alias for {@link physics.GRAVITATIONAL}
 96  * @type bigrat
 97  * @static
 98  * @final
 99  */
100 physics.G = physics.GRAVITATIONAL;

102 /**
103  * The speed of light in vacuum (c) in m / s
104  *
105  * @property SPEEDOFLIGHT
106  * @type bigrat
107  * @static
108  * @final
109  */

112 /**
113  * Alias for {@link physics.SPEEDOFLIGHT}
