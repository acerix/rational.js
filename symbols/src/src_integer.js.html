 21 /**
 22  * @class Integer
 23  * @name integer
 24  */
 25 var integer = {};
 26 
 27 /**
 28  * Find the greatest common divisor of two integers
 29  *
 30  * @param {Integer} a the first operand
 31  * @param {Integer} b the second operand
 32  * @returns {Integer} greatest common divisor
 33  */
 34 integer.greatest_common_divisor = function(a, b) {
 35 	if (b===1 || a===1) return 1;
 36 	var t;
 37 	while (b !== 0) {
 38 		t = b;
 39 		b = a % b;
 40 		if (!isFinite(b)) return 0;
 41 		a = t;
 42 	}
 43 	return a;
 44 }
 45 
 46 /**
 47  * Alias for {@link integer.greatest_common_divisor}
 48  * @function
 49  */
 50 integer.gcd = integer.greatest_common_divisor;
 51 
 52 /**
 53  * Returns a new random unsigned integer with a maximum size specified in bytes
 54  *
 55  * @param {Integer} maximum number of bits
 56  * @returns {Integer} a random integer from Zero to 2^bytes (inclusive)
 57  */
 58 integer.fromRandom = function(bits) {
 59 	return Math.random() * (1 << bits) >>> 0;
 60 };
 61 
 62 /**
 63  * Returns a new unsigned integer with the number of milliseconds since the Unix epoch
 64  *
 65  * @returns {Integer} milliseconds since the 60's ended
 66  */
 67 integer.fromMillitime = function() {
 68 	return +new Date;
 69 };
 70 
 71 /**
 72  * Returns a new unsigned integer with the number of milliseconds since the Unix epoch
 73  *
 74  * @returns {Integer} milliseconds since the 60's ended
 75  */
 76 integer.fromUnixtime = function() {
 77 	return integer.fromMillitime() / 1000 | 0;
 78 };
