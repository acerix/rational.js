 21 /**
 22  * @class Alpha to Omega
 23  * @name alpha
 24  */
 25 var alpha = {};
 26 
 27 /**
 28  * List of some distinct looking greek letters for use in polynomials
 29  *
 30  * @property GREEK
 31  * @type Array
 32  * @static
 33  * @final
 34  */
 35 alpha.GREEK = 'αβγδμπρςτφχψωΘΦΩ'.split('');
 36 
 37 /**
 38  * Returns the next consecutive greek letter from the list
 39  *
 40  * @returns {String} greek letter
 41  */
 42 alpha.iterator = function() {
 43 	this.i = 0;
 44 	this.next = function() {
 45 		return this.i < alpha.GREEK.length ? alpha.GREEK[this.i++] : '?';
 46 	}
 47 };
 48 
 49 if(typeof(exports) !== 'undefined') {
 50 	exports.alpha = alpha;
 51 }
