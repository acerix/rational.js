 26 /**
 27  * Chainable wrapper functions for rat.js
 28  *
 29  * @class Rational Number (Chainable)
 30  * @name rational
 31  * @requires rat
 32  */
 33 var rational = function(numerator, denominator) {
 34 	//if (numerator.constructor.name==='rational'||numerator.denominator.name==='rational') return rational.divide(numerator, denominator);
 35 	this.a = rat.fromValues(parseInt(numerator), parseInt(denominator));
 36 }
 37 
 38 /**
 39  * Expresses the rational in a string as "numerator/denominator"
 40  *
 41  * @returns {String} numerator/denominator
 42  */
 43 rational.prototype.toString = function() {
 44 	return rat.str(this.a);
 45 };
 46 
 47 /**
 48  * Returns a rational from an array of integers representing a continued fraction
 49  *
 50  * @param {Integer} maximum number of iterations
 51  * @returns {Array} integers of the continued fraction
 52  */
 53 rational.prototype.toContinuedFraction = function(loop_limit) {
 54 	return rat.toContinuedFraction(this.a, loop_limit);
 55 };

 72 /**
 73  * Add
 74  *
 75  * @param {rational} b the second operand
 76  * @returns {rational} out
 77  */
 78 rational.prototype.add = function(b) {
 79 	var out = rat.create();
 80 	rat.add(out, this.a, b.a);
 81 	return new rational(out[0], out[1]);
 82 };

219 /**
220  * Returns a rational from a rat array
221  *
222  * @param {rat} rat
223  * @returns {rational} out
224  */
225 rational.fromRat = function (a) {
226 	var r = new rational();
227 	r.a[0] = a[0];
228 	r.a[1] = a[1];
229 	return r;
230 };

232 /**
233  * Returns a rational from an integer
234  *
235  * @param {Integer} signed integer
236  * @returns {rational} out
237  */
238 rational.fromInteger = function (a) {
239 	return rational.fromRat(rat.fromInteger(a));
240 };
