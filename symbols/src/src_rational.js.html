 21 /**
 22  * Chainable wrapper functions for rat.js
 23  *
 24  * @class Rational Number (Chainable)
 25  * @name rational
 26  * @requires integer rat
 27  */
 28 var rational = function(numerator, denominator) {
 29 	var self = this;
 30 	if (numerator.constructor.name==='rational'||numerator.denominator.name==='rational')
 31 		return rational.divide(numerator, denominator);
 32 	self.a = rat.fromValues(parseInt(numerator), parseInt(denominator));
 33 	return self.a;
 34 }
 35 
 36 /**
 37  * Inverts a rational
 38  *
 39  * @returns {rational} out
 40  */
 41 rational.invert = function() {
 42 	return rat.invert(self.a);
 43 };
 44

 57 rational.add = function(b) {
 58 	return rat.add(rat.create(), self.a, b);
