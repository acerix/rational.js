 21 /**
 22  * Chainable wrapper functions for rat.js
 23  *
 24  * @class Rational Number (Chainable)
 25  * @name rational
 26  * @requires integer rat
 27  */
 28 var rational = function(numerator, denominator) {
 29 	//if (numerator.constructor.name==='rational'||numerator.denominator.name==='rational') return rational.divide(numerator, denominator);
 30 	this.a = rat.fromValues(parseInt(numerator), parseInt(denominator));
 31 }
 32 
 33 /**
 34  * Expresses the rational in a string as "numerator/denominator"
 35  *
 36  * @returns {String} numerator/denominator
 37  */
 38 rational.prototype.toString = function() {
 39 	return rat.str(this.a);
 40 };
 41 
 42 /**
 43  * Inverts a rational
 44  *
 45  * @returns {rational} out
 46  */
 47 rational.prototype.invert = function() {
 48 	return rat.invert(this.a);
 49 };
 50 
 51 /**
 52  * Alias for {@link rational.invert}
 53  * @function
 54  */
 55 rational.prototype.reciprocal = rational.prototype.invert;

 57 /**
 58  * Add
 59  *
 60  * @param {rational} b the second operand
 61  * @returns {rational} out
 62  */
 63 rational.prototype.add = function(b) {
 64 	var out = rat.create();
 65 	rat.add(out, this.a, b.a);
 66 	return new rational(out[0], out[1]);
 67 };

186 /**
187  * Returns a rational from a rat array
188  *
189  * @param {rat} rat
190  * @returns {rational} out
191  */
192 rational.fromRat = function (a) {
193 	var r = new rational();
194 	r.a[0] = a[0];
195 	r.a[1] = a[1];
196 	return r;
197 };
198 
199 /**
200  * Returns a rational from an integer
201  *
202  * @param {Integer} signed integer
203  * @returns {rational} out
204  */
205 rational.prototype.fromInteger = function (a) {
206 	return new rational.fromRat(rat.fromInteger(a));
207 };
208 
209 /**
210  * Returns a rational from the inverse of an integer
211  *

213  * @returns {rational} out
214  */
215 rational.prototype.fromIntegerInverse = function (a) {
