 22 /**
 23  * @class Arbitrary Sized Integer
 24  * @name bigint
 25  * @requires BigInteger
 26  */
 27 var bigint = {};
 28 
 29 /**
 30  * Find the greatest common divisor of two big integers
 31  *
 32  * @param {BigInteger} a the first operand
 33  * @param {BigInteger} b the second operand
 34  * @returns {BigInteger} greatest common divisor
 35  */
 36 bigint.greatest_common_divisor = function(a, b) {
 37 	if (b.isUnit() || a.isUnit()) return BigInteger.ONE;
 38 	var t;
 39 	while (!b.isZero()) {
 40 		t = b;
 41 		b = a.remainder(b);
 42 		a = t;
 43 	}
 44 	return a;
 45 }

 47 /**
 48  * Creates a new, empty bigint
 49  *
 50  * @returns {bigint} a new bigint
 51  */
 52 bigint.create = function() {
 53 	return [0];
 54 };
 55 
 56 /**
 57  * Creates a new bigint initialized with values from an existing number
 58  *
 59  * @param {bigint} a number to clone
 60  * @returns {bigint} a new bigintional number
 61  */
 62 bigint.clone = function(a) {
 63 	var out = [];
 64 	for (var i=0, l=a.length; i<l; i++)
 65 		out[i] = a[i];
 66 	return out;
 67 };
 68 
 69 /**
 70  * Copy the values from one bigint to another
 71  *
 72  * @param {bigint} out the receiving number
 73  * @param {bigint} a the source number
 74  * @returns {bigint} out
 75  */
 76 bigint.copy = function(out, a) {
 77 	out = [];
 78 	for (var i=0, l=a.length; i<l; i++)
 79 		out[i] = a[i];
 80 	return out;
 81 };
 82 
 83 /**
 84  * Absolute value of a bigint
 85  *
 86  * @param {bigint} out the receiving number
 87  * @param {bigint} a number to take the absolute value of
 88  * @returns {bigint} out
 89  */
 90 bigint.abs = function(out, a) {
 91 	out[0] = Math.abs(a[0]);
 92 	return out;
 93 };
 94 
 95 /**
 96  * Returns a string representation
 97  *
 98  * @param {bigint} a number to represent as a string
 99  * @returns {String} string representation of the number
100  */
101 bigint.str = function (a) {
102 	return a[0].toString();
103 };
104 
105 /**
106  * Returns an integer to it's native type
107  *
108  * @param {bigint} a number to return as an integer
109  * @returns {Integer} an integer or Infinity when the max size is exceeded
110  */
111 bigint.toInteger = function (a) {
112 	return out[0];
113 };
114 
115 /**
116  * Returns a bigint from an integer
117  *
118  * @param {bigint} out the receiving number
119  * @param {Integer} integer
120  * @returns {bigint} out
121  */
122 bigint.fromInteger = function (a) {
123 	return [parseInt(a)];
124 };
125 
126 /**
127  * Zero, the additive identity
128  *
129  * @property bigint_ZERO
130  * @type bigint_ARRAY_TYPE
131  * @static
132  * @final
133  */
134 var BIGINT_ZERO = bigint.fromInteger(0);
135 
136 /**
137  * One, the multiplicative identity
138  *
139  * @property bigint_ONE
140  * @type bigint_ARRAY_TYPE
141  * @static
142  * @final
143  */
144 var BIGINT_ONE = bigint.fromInteger(1);
