 27 /**
 28  * @class Arbitrary Sized Integer
 29  * @name bigint
 30  * @requires BigInteger
 31  */
 32 var bigint = {};
 33 
 34 /**
 35  * Find the greatest common divisor of two big integers
 36  *
 37  * @param {BigInteger} a the first operand
 38  * @param {BigInteger} b the second operand
 39  * @returns {BigInteger} greatest common divisor
 40  */
 41 bigint.greatest_common_divisor = function(a, b) {
 42 	if (b.isUnit() || a.isUnit()) return BigInteger.ONE;
 43 	var t;
 44 	while (!b.isZero()) {
 45 		t = b;
 46 		b = a.remainder(b);
 47 		a = t;
 48 	}
 49 	return a;
 50 }

 52 /**
 53  * Creates a new, empty bigint
 54  *
 55  * @returns {bigint} a new bigint
 56  */
 57 bigint.create = function() {
 58 	return [0];
 59 };
 60 
 61 /**
 62  * Creates a new bigint initialized with values from an existing number
 63  *
 64  * @param {bigint} a number to clone
 65  * @returns {bigint} a new bigintional number
 66  */
 67 bigint.clone = function(a) {
 68 	var out = [];
 69 	for (var i=0, l=a.length; i<l; i++)
 70 		out[i] = a[i];
 71 	return out;
 72 };
 73 
 74 /**
 75  * Copy the values from one bigint to another
 76  *
 77  * @param {bigint} out the receiving number
 78  * @param {bigint} a the source number
 79  * @returns {bigint} out
 80  */
 81 bigint.copy = function(out, a) {
 82 	out = [];
 83 	for (var i=0, l=a.length; i<l; i++)
 84 		out[i] = a[i];
 85 	return out;
 86 };
 87 
 88 /**
 89  * Absolute value of a bigint
 90  *
 91  * @param {bigint} out the receiving number
 92  * @param {bigint} a number to take the absolute value of
 93  * @returns {bigint} out
 94  */
 95 bigint.abs = function(out, a) {
 96 	out[0] = Math.abs(a[0]);
 97 	return out;
 98 };
 99 
100 /**
101  * Returns a string representation
102  *
103  * @param {bigint} a number to represent as a string
104  * @returns {String} string representation of the number
105  */
106 bigint.str = function (a) {
107 	return a[0].toString();
108 };
109 
110 /**
111  * Returns an integer to it's native type
112  *
113  * @param {bigint} a number to return as an integer
114  * @returns {Integer} an integer or Infinity when the max size is exceeded
115  */
116 bigint.toInteger = function (a) {
117 	return out[0];
118 };
119 
120 /**
121  * Returns a bigint from an integer
122  *
123  * @param {bigint} out the receiving number
124  * @param {Integer} integer
125  * @returns {bigint} out
126  */
127 bigint.fromInteger = function (a) {
128 	return [parseInt(a)];
129 };

131 /**
132  * Zero, the additive identity
133  *
134  * @property ZERO
135  * @type bigint
136  * @static
137  * @final
138  */
139 bigint.ZERO = bigint.fromInteger(0);

141 /**
142  * One, the multiplicative identity
143  *
144  * @property ONE
145  * @type bigint
146  * @static
147  * @final
148  */
149 bigint.ONE = bigint.fromInteger(1);
