 28 /**
 29  * Creates a new, empty polyrat
 30  *
 31  * @returns {polyrat} a new polynumber
 32  */
 33 polyrat.create = function() {

 35 };

 37 /**
 38  * Creates a new polynumber initialized with the given values
 39  *
 40  * @param {Array} 2D array of values
 41  * @returns {polyrat} a new polynumber
 42  */
 43 polyrat.fromValues = function(a) {
 44 	var out = [];
 45 	for (var i in a) {
 46 		out[i] = [];
 47 		for (var j in a[i]) {
 48 			//out[i][j] = rat.fromInteger(a[i][j]);
 49 		}
 50 	}
 51 	return out;
 52 };
 53 
 54 /**
 55  * Returns a string representation
 56  *
 57  * @param {rat} a rational polynumber to represent as a string
 58  * @returns {String} string representation of the number
 59  */
 60 polyrat.str = function (a) {
 61 	return a[0][0][0].toString();
 62 };
 63 
 64 /**
 65  * Zero, the additive identity
 66  *
 67  * @property POLYRAT_ZERO
 68  * @type Array
 69  * @static
 70  * @final
 71  */
 72 var POLYRAT_ZERO = polyrat.fromValues([[0]]);
 73 
 74 /**
 75  * One, the multiplicative identity
 76  *
 77  * @property RAT_ONE
 78  * @type RAT_ARRAY_TYPE
 79  * @static
 80  * @final
 81  */
 82 var POLYRAT_IDENTITY = polyrat.fromValues([[1]]);
