 88 /**
 89  * Add another variable/dimension to a polynumber
 90  *
 91  * @param {polyrat} out the receiving number
 92  * @param {rat} a the first operand
 93  * @param {Array} array of values (of type "rat")
 94  * @returns {polyrat} a new polynumber
 95  */
 96 polyrat.mergeDimension = function(out, a, m) {
 97 	out[0] = [a.slice()];
 98 	out[1].push(0);
 99 	for (var i in m) out[0].push(m[i]);
100 	return out;

103 /**
104  * Evaluate a polynomial for a certain variable(s)
105  *
106  * @param {bigrat} out the receiving number
107  * @param {polyrat} a the polynumber to evaluate 
108  * @param {Array} array of values (of type "bigrat") to plug in to the polynomial
109  * @returns {bigrat} resulting big rational number
110  */
111 polyrat.evaluate = function(out, a, m) {
112 	//var result = 0;
113 	bigrat.copy(out, bigrat.ZERO);
114 	var d = polyrat.countDimensions(a);
115 	if (d===0) {
116 		return '0';
117 	}
118 	else if (d===1) {
119 		for (var i in a[0]) {
120 			if (!a[0][i]) continue;
121 			//result += a[0][i] * Math.pow(m[0], a[1][0] + +i);
122 			var p = bigrat.fromInteger(a[0][i]);
123 			var pi = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[0]), a[1][0] + +i);
124 			bigrat.mul(p, p, pi);
125 			bigrat.add(out, out, p);
126 		}
127 	}
128 	else if (d===2) {
129 		for (var j in a[0]) {
130 			for (var i in a[0][j]) {
131 				if (!a[0][j][i]) continue;
132 				//result += a[0][j][i] * Math.pow(m[0], a[1][1] + +i) * Math.pow(m[1], a[1][0] + +j);
133 				var p = bigrat.fromInteger(a[0][j][i]);
134 				var pi = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[0]), a[1][1] + +i);
135 				var pj = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[1]), a[1][0] + +j);
136 				bigrat.mul(p, p, pi);
137 				bigrat.mul(p, p, pj);
138 				bigrat.add(out, out, p);
139 			}
140 		}
141 	}
142 	else if (d===3) {
143 		for (var k in a[0]) {
144 			for (var j in a[0][k]) {
145 				for (var i in a[0][k][j]) {
146 					if (!a[0][k][j][i]) continue;
147 					//result += a[0][k][j][i] * Math.pow(m[0], a[1][2] + +i) * Math.pow(m[1], a[1][1] + +j) * Math.pow(m[2], a[1][0] + +k);
148 					var p = bigrat.fromInteger(a[0][k][j][i]);
149 					var pi = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[0]), a[1][2] + +i);
150 					var pj = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[1]), a[1][1] + +j);
151 					var pk = bigrat.pow(bigrat.create(), bigrat.fromInteger(m[1]), a[1][0] + +k);
152 					bigrat.mul(p, p, pi);
153 					bigrat.mul(p, p, pj);
154 					bigrat.mul(p, p, pk);
155 					bigrat.add(out, out, p);
156 				}
157 			}
158 		}
159 	}
160 	else {
161 		// recursive function?
162 		alert('unsupported dimension '+d);
163 	}
164 	//return out = bigrat.fromInteger(parseInt(result));
165 	return out;
166 };

206 /**
207  * Returns a string representation (polynomial format, in ascending order by degree, then alphabetically) -- not yet!
208  *
209  * @param {polyrat} a rational polynumber to represent as a string
210  * @returns {String} string representation of the number
211  */
212 polyrat.str = function(a) {
213 	var s = '';
214 	var vars = new alpha.iterator;
215 	
216 	var _i = vars.next();
217 	var _j = vars.next();
218 	var _k = vars.next();
219 	
220 	var d = polyrat.countDimensions(a);
221 	if (d===0) {
222 		s += a.toString();
223 	}
224 	else if (d===1) {
225 		for (var i in a[0]) {
226 			if (!a[0][i]) continue;
227 			if (s) s += ' ' + (a[0][i]<0?'-':'+') + ' ';
228 			else if (a[0][i]<0) s += '-';
229 			var c = Math.abs(a[0][i]);
230 			var i_power = a[1][0] + +i;
231 			if (c!==1||(i_power===0)) s += c;
232 			if (i_power!==0) s += _i + (i_power === 1 ? '' : '<sup>'+i_power+'</sup>');
233 		}
234 	}
235 	else if (d===2) {
236 		for (var j in a[0]) {
237 			for (var i in a[0][j]) {
238 				if (!a[0][j][i]) continue;
239 				if (s) s += ' ' + (a[0][j][i]<0?'-':'+') + ' ';
240 				else if (a[0][j][i]<0) s += '-';
241 				var c = Math.abs(a[0][j][i]);
242 				var i_power = a[1][1] + +i;
243 				var j_power = a[1][0] + +j;
244 				if (c!==1||(j_power===0&&i_power===0)) s += c;
245 				if (i_power!==0) s += _i + (i_power === 1 ? '' : '<sup>'+i_power+'</sup>');
246 				if (j_power!==0) s += _j + (j_power === 1 ? '' : '<sup>'+j_power+'</sup>');
247 			}
248 		}
249 	}
250 	else if (d===3) {
251 		for (var k in a[0]) {
252 			for (var j in a[0][k]) {
253 				for (var i in a[0][k][j]) {
254 					if (!a[0][k][j][i]) continue;
255 					if (s) s += ' ' + (a[0][k][j][i]<0?'-':'+') + ' ';
256 					else if (a[0][k][j][i]<0) s += '-';
257 					var c = Math.abs(a[0][k][j][i]);
258 					var i_power = a[1][2] + +i;
259 					var j_power = a[1][1] + +j;
260 					var k_power = a[1][0] + +k;
261 					if (c!==1||(k_power===0&&j_power===0&&i_power===0)) s += c;
262 					if (i_power!==0) s += _i + (i_power === 1 ? '' : '<sup>'+i_power+'</sup>');
263 					if (j_power!==0) s += _j + (j_power === 1 ? '' : '<sup>'+j_power+'</sup>');
264 					if (k_power!==0) s += _k + (k_power === 1 ? '' : '<sup>'+k_power+'</sup>');
265 				}
266 			}
267 		}
268 	}
269 	else {
270 		alert('unsupported dimension '+d);
271 	}
272 	if (s==='') s = '0';
273 	return s;
274 };

277 /**
278  * Evaluate a polynomial for a certain variable(s)
279  *
280  * @param {polyrat} a polynumber
281  * @returns {String} javascript formula to evaluate the polynomial
282  */
283 polyrat.getJSFormula = function(a) {
284 	var f = '';
285 	var d = polyrat.countDimensions(a);
286 	if (d===0) {
287 		return '0';
288 	}
289 	else if (d===1) {
290 		for (var i in a[0]) {
291 			if (!a[0][i]) continue;
292 			if (f) f += '+';
293 			f += a[0][i];
294 			var i_power = a[1][0] + +i;
295 			if (i_power===0) continue;
296 			else if (i_power===1) f += '*m[0]';
297 			else f += '*Math.pow(m[0],' + i_power + ')';
298 		}
299 	}
300 	else if (d===2) {
301 		for (var j in a[0]) {
302 			for (var i in a[0][j]) {
303 				if (!a[0][j][i]) continue;
304 				if (f) f += '+';
305 				f += a[0][j][i];
306 				var i_power = a[1][1] + +i;
307 				var j_power = a[1][0] + +j;
308 				if (i_power!==0) {
309 					if (i_power===1) f += '*m[0]';
310 					else f += '*Math.pow(m[0],' + i_power + ')';
311 				}
312 				if (j_power!==0) {
313 					if (j_power===1) f += '*m[1]';
314 					else f += '*Math.pow(m[1],' + j_power + ')';
315 				}
316 			}
317 		}
318 	}
319 	else if (d===3) {
320 		for (var k in a[0]) {
321 			for (var j in a[0][k]) {
322 				for (var i in a[0][k][j]) {
323 					if (!a[0][k][j][i]) continue;
324 					if (f) f += '+';
325 					f += a[0][k][j][i];
326 					var i_power = a[1][2] + +i;
327 					var j_power = a[1][1] + +j;
328 					var k_power = a[1][0] + +k;
329 					if (i_power!==0) {
330 						if (i_power===1) f += '*m[0]';
331 						else f += '*Math.pow(m[0],' + i_power + ')';
332 					}
333 					if (j_power!==0) {
334 						if (j_power===1) f += '*m[1]';
335 						else f += '*Math.pow(m[1],' + j_power + ')';
336 					}
337 					if (k_power!==0) {
338 						if (k_power===1) f += '*m[2]';
339 						else f += '*Math.pow(m[2],' + k_power + ')';
340 					}
341 				}
342 			}
343 		}
344 	}
345 	else {
346 		alert('unsupported dimension '+d);
347 	}
348 	if (f==='') f = '0';
349 	return f;
350 };

355 /**
356  * Evaluate a polynomial for a certain variable(s), modified to work in GLSL (to run on GPU)
357  *
358  * @param {polyrat} a polynumber
359  * @returns {String} GLSL formula to evaluate the polynomial
360  */
361 polyrat.getGLSLFormula = function(a) {
362 	var f = '';
363 	var d = polyrat.countDimensions(a);
364 	if (d===0) {
365 		return '0.0';
366 	}
367 	else if (d===1) {
368 		for (var i in a[0]) {
369 			if (!a[0][i]) continue;
370 			if (f) f += '+';
371 			f += a[0][i]+'.0';
372 			var i_power = a[1][0] + +i;
373 			if (i_power===0) continue;
374 			else if (i_power===1) f += '*m_0';
375 			else if (i_power>1) f += ('*m_0').repeat(i_power);
376 			else f += '*pow(1.0'+('*m_0').repeat(Math.abs(i_power))+',-1.0)';
377 		}
378 	}
379 	else if (d===2) {
380 		for (var j in a[0]) {
381 			for (var i in a[0][j]) {
382 				if (!a[0][j][i]) continue;
383 				if (f) f += '+';
384 				f += a[0][j][i]+'.0';
385 				var i_power = a[1][1] + +i;
386 				var j_power = a[1][0] + +j;
387 				if (i_power!==0) {
388 					if (i_power===1) f += '*m_0';
389 					else if (i_power>1) f += ('*m_0').repeat(i_power);
390 					else f += '*pow(1.0'+('*m_0').repeat(Math.abs(i_power))+',-1.0)';
391 				}
392 				if (j_power!==0) {
393 					if (j_power===1) f += '*m_1';
394 					else if (j_power>1) f += ('*m_1').repeat(j_power);
395 					else f += '*pow(1.0'+('*m_1').repeat(Math.abs(j_power))+',-1.0)';
396 				}
397 			}
398 		}
399 	}
400 	else {
401 		alert('unsupported dimension '+d);
402 	}
403 	if (f==='') f = '0.0';
404 	return f;
405 };

438 /**
439  * Adds two polyrats
440  *
441  * @param {polyrat} out the receiving number
442  * @param {polyrat} a the first operand
443  * @param {polyrat} b the second operand
444  * @returns {polyrat} out
445  */
446 polyrat.add = function(out, a, b) {
447 	var 	a_dimensions = polyrat.countDimensions(a),
448 			b_dimensions = polyrat.countDimensions(b);
449 	if (a[1][i]!==b[1][i]) {
450 		alert('adding polyrats only works if the offsets are the same (for now)');
451 	}
452 	if (a_dimensions===1&&b_dimensions===1) {
453 		for (var i in a[0]) {
454 			out[0][i] = typeof b[0][i] === 'undefined' ? a[0][i] : a[0][i] + b[0][i];
455 		}
456 		for (var i in b[0]) {
457 			if (typeof a[0][i] === 'undefined') out[0][i] = b[0][i];
458 		}
459 		for (var i in out[0]) {
460 			if (typeof a[0][i] === 'undefined' && typeof b[0][i] === 'undefined') out[0][i] = null;
461 		}
462 	}
463 	else {
464 		alert('unsupported dimension '+a_dimensions);
465 	}
466 	var i=a_dimensions;
467 	while (i--) out[1][i] = a[1][i];
468 	return out;
469 };
