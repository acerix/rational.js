 21 /**
 22  * @class Rational Polynumber (of arbitrary dimension)
 23  * @name polyrat
 24  * @requires rat alpha
 25  */
 26 var polyrat = {};
 27 
 28 /**
 29  * Creates a new, empty polyrat
 30  *
 31  * @returns {polyrat} a new polynumber
 32  */
 33 polyrat.create = function() {
 34 	return [
 35 		[], // multi-dimensional array of coefficients (will be rational numbers, for now integers...)
 36 		[] // array of coefficient offsets (integers)
 37 	];
 38 };
 39 
 40 /**
 41  * Creates a new polynumber initialized with the given array of either integers or rats
 42  *
 43  * @param {Array} array of values (of type "rat")
 44  * @returns {polyrat} a new polynumber
 45  */
 46 polyrat.fromValues = function(a) {
 47 	var out = polyrat.create();
 48 	out[0] = a.slice();
 49 	out[1] = new Array(polyrat.countDimensions(out));
 50 	for (var i=out[1].length; i--; 1) out[1][i] = 0;
 51 	return polyrat.normalize(out, out);
 52 };
 53 
 54 /**
 55  * Add another variable/dimension to a polynumber
 56  *
 57  * @param {polyrat} out the receiving number
 58  * @param {rat} a the first operand
 59  * @param {Array} array of values (of type "rat")
 60  * @returns {polyrat} a new polynumber
 61  */
 62 polyrat.mergeDimension = function(out, a, m) {
 63 	out[0] = [a.slice()];
 64 	out[1].push(0);
 65 	for (var i in m) out[0].push(m[i]);
 66 	return out;
 67 };

134 /**
135  * Returns a string representation (polynomial format, in ascending order by degree, then alphabetically) -- not yet!
136  *
137  * @param {polyrat} a rational polynumber to represent as a string
138  * @returns {String} string representation of the number
139  */
140 polyrat.str = function(a) {
141 	var s = '';
142 	var vars = new alpha.iterator;
143 	
144 	var _i = vars.next();
145 	var _j = vars.next();
146 	var _k = vars.next();
147 	
148 	var d = polyrat.countDimensions(a);
149 	if (d===0) {
150 		s += a.toString();
151 	}
152 	else if (d===1) {
153 		for (var i in a[0]) {
154 			if (!a[0][i]) continue;
155 			if (s) s += ' ' + (a[0][i]<0?'-':'+') + ' ';
156 			else if (a[0][i]<0) s += '-';
157 			var c = Math.abs(a[0][i]);
158 			if (c!=1||(i==0)) s += c;
159 			if (i!=0) s += _i + (i > 1 ? '<sup>'+i+'</sup>' : '');
160 		}
161 	}
162 	else if (d===2) {
163 		for (var j in a[0]) {
164 			for (var i in a[0][j]) {
165 				if (!a[0][j][i]) continue;
166 				if (s) s += ' ' + (a[0][j][i]<0?'-':'+') + ' ';
167 				else if (a[0][j][i]<0) s += '-';
168 				var c = Math.abs(a[0][j][i]);
169 				if (c!=1||(j==0&&i==0)) s += c;
170 				if (i!=0) s += _i + (i > 1 ? '<sup>'+i+'</sup>' : '');
171 				if (j!=0) s += _j + (j > 1 ? '<sup>'+j+'</sup>' : '');
172 			}
173 		}
174 	}
175 	else if (d===3) {
176 		for (var k in a[0]) {
177 			for (var j in a[0][k]) {
178 				for (var i in a[0][k][j]) {
179 					if (!a[0][k][j][i]) continue;
180 					if (s) s += ' ' + (a[0][k][j][i]<0?'-':'+') + ' ';
181 					else if (a[0][k][j][i]<0) s += '-';
182 					var c = Math.abs(a[0][k][j][i]);
183 					if (c!=1||(k==0&&j==0&&i==0)) s += c;
184 					if (i!=0) s += _i + (i > 1 ? '<sup>'+(a[1][2]+ +i)+'</sup>' : '');
185 					if (j!=0) s += _j + (j > 1 ? '<sup>'+(a[1][1]+ +j)+'</sup>' : '');
186 					if (k!=0) s += _k + (k > 1 ? '<sup>'+(a[1][0]+ +k)+'</sup>' : '');
187 				}
188 			}
189 		}
190 	}
191 	else {
192 		// recursive function?
193 		alert('unsupported dimension');
194 	}
195 	if (s==='') s = '0';
196 	return s;
197 };

232 /**
233  * Count the number of dimensions of an array
234  *
235  * @param {polyrat} a number to count the dimensions of
236  * @returns {Integer} number of dimensions
237  */
238 polyrat.countDimensions = function(a) {
239 	return typeof a[0] === 'object' ? polyrat._countDimensions(a[0], 1) : 0;
240 };
241 
242 /**
243  * Count the number of dimensions of an array recursively
244  *
245  * @param {polyrat} an array to count the dimensions of
246  * @param {polyrat} c (counter used during recursion)
247  * @returns {Integer} number of dimensions
248  */
249 polyrat._countDimensions = function(a, c) {
250 	return typeof a[0] === 'object' ? polyrat._countDimensions(a[0], ++c) : c;
251 };
