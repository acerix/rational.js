 21 /**
 22  * @class Planet physics
 23  * @name planet
 24  * @requires vec3
 25  */
 26 var planet = function(x, y, radius, mass) {
 27 	
 28 	// position
 29 	this.p = vec3.fromValues(x, y, 0);
 30 	
 31 	// velocity
 32 	this.v = vec3.fromValues(0, 0, 0);
 33 	
 34 	// quadrance (radius squared)
 35 	//this.R = bigrat.pow(bigrat.create(), bigrat.fromDecimal(radius), 2);
 36 	this.R = radius * radius;
 37 	
 38 	// mass
 39 	//this.m = bigrat.fromDecimal(mass);
 40 	this.m = mass;
 41 }
 42 
 43 /**
 44  * Returns a string representation of the planet
 45  *
 46  * @returns {String}
 47  */
 48 planet.prototype.toString = function() {
 49 	return vec3.str(this.p);
 50 };
 51 
 52 /**
 53  * Returns the force of gravity pulling this planet towards that one
 54  *
 55  * @returns {vec3}
 56  */
 57 planet.prototype.getGravity = function(that) {
 58 	
 59 	var gravity = vec3.create();
 60 	vec3.sub(gravity, this.p, that.p);
 61 	vec3.normalize(gravity, gravity);
 62 	
 63 	var magnitude = (this.m * that.m) / vec3.sqrDist(this.p, that.p);
 64 	
 65 	return vec3.scale(gravity, gravity, magnitude);
 66 };

 69 	exports.planet = planet;
