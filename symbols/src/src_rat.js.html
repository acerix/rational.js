163 /**
164  * Adds two rats
165  *
166  * @param {rat} out the receiving number
167  * @param {rat} a the first operand
168  * @param {rat} b the second operand
169  * @returns {rat} out
170  */
171 rat.add = function(out, a, b) {
172 	if (a[1]===b[1]) {
173 		out[0] = a[0] + b[0];
174 		out[1] = a[1];
175 	}
176 	else {
177 		out[0] = a[0] * b[1] + b[0] * a[1];
178 		out[1] = a[1] * b[1];
179 	}
180 	return rat.normalize(out, out);
181 };

183 /**
184  * Subtracts two rats
185  *
186  * @param {rat} out the receiving number
187  * @param {rat} a the first operand
188  * @param {rat} b the second operand
189  * @returns {rat} out
190  */
191 rat.subtract = function(out, a, b) {
192 	if (a[1]===b[1]) {
193 		out[0] = a[0] - b[0];
194 		out[1] = a[1];
195 	}
196 	else {
197 		out[0] = a[0] * b[1] - b[0] * a[1];
198 		out[1] = a[1] * b[1];
199 	}
200 	return rat.normalize(out, out);
201 };

274 /**
275  * Returns true when the first rat is approximately equal to the second
276  *
277  * @param {rat} a the first operand
278  * @param {rat} b the second operand
279  * @returns {Bool} true when the difference between the two rats is less than rat.INFINITESIMAL
280  */
281 rat.approximates = function(a, b) {
282 	if (rat.equals(a, b)) return true;
283     var d = rat.create();
284     rat.sub(d, a, b);
285     rat.abs(d, d);
286     return rat.isLessThan(d, rat.INFINITESIMAL);
287 };

390 /**
391  * Normalize a rat
392  *
393  * @param {rat} out the receiving number
394  * @param {rat} a number to normalize
395  * @returns {rat} out
396  */
397 rat.normalize = function(out, a) {
398 	if (isNaN(a[0])||isNaN(a[1])) return out = rat.clone(rat.INFINULL);
399 	if (a[1] >= 0) {
400 		out[0] = a[0];
401 		out[1] = a[1];
402 		if (a[1] === 0) return out = a;
403 	}
404 	else {
405 		out[0] = -a[0];
406 		out[1] = -a[1];
407 	}
408 	var gcd = integer.greatest_common_divisor(Math.abs(out[0]), out[1]);
409 	if (gcd > 1) {
410 		out[0] /= gcd;
411 		out[1] /= gcd;
412 	}
413 	return out;
414 };

435 /**
436  * Raises a rat to an integer exponent
437  *
438  * @param {rat} out the receiving number
439  * @param {rat} a the number to exponentiate
440  * @param {Integer} p power to raise the number by
441  * @returns {rat} out
442  */
443 rat.power = function(out, a, p) {
444 	if (p>0) {
445 		out[0] = Math.pow(a[0], p);
446 		out[1] = Math.pow(a[1], p);
447 	}
448 	else if (p<0) {
449 		p = Math.abs(p);
450 		out[0] = Math.pow(a[1], p);
451 		out[1] = Math.pow(a[0], p);
452 	}
453 	else {
454 		rat.copy(out, rat.ONE);
455 	}
456 	return out;
457 };

476 /**
477  * Find a rat approximation which equals the input rat when raised to the given integer exponent
478  * 
479  * Newton's method converges alot faster... could that be used to find the pattern in the SB tree?
480  *
481  * @param {rat} out the receiving number
482  * @param {rat} a the number to find the root of
483  * @param {Integer} n
484  * @returns {rat} out
485  */
486 rat.nthRoot = function (out, a, n) {
487 	
488 	if (a===rat.ZERO) return rat.copy(out, rat.ZERO);
489 	if (a===rat.ONE) return rat.copy(out, rat.ONE);
490 	if (a===rat.INFINITY) return rat.copy(out, rat.INFINITY);
491 	if (a===rat.INFINULL) return rat.copy(out, rat.INFINULL);
492 	
493 	var neg = rat.isNegative(a);
494 	if (neg) a[0] = -a[0];
495 	
496 	out = rat.copy(out, rat.ONE);
497 	var m = [1, 0, 0, 1];
498 	var test = rat.clone(rat.ONE);
499 	
500 	var c = RAT_MAX_LOOPS;
501 	while ( !rat.approximates(a, test) && c-- ) {
502 		if (rat.isLessThan(a, test)) {
503 			m[0] += m[1];
504 			m[2] += m[3];
505 		}
506 		else {
507 			m[1] += m[0];
508 			m[3] += m[2];
509 		}
510 		out[0] = m[0] + m[1];
511 		out[1] = m[2] + m[3];
512 		rat.pow(test, out, n);
513 	}
514 	
515 	if (neg) { 
516 		a[0] = -a[0];
517 		if (n%2===1) rat.neg(out, out);
518 	}
519 		
520 	return out;
521 };

652 /**
653  * Returns a rat from a decimal number, copying to an existing rat
654  *
655  * @param {rat} out the receiving number
656  * @param {Number} a decimal number
657  * @returns {rat} out
658  */
659 rat.fromDecimal_copy = function (out, a) {
660 	
661 	a = parseFloat(a);
662 	if (a===0) return rat.copy(out, rat.ZERO);
663 	if (a===1) return rat.copy(out, rat.ONE);
664 	if (a===Infinity) return rat.copy(out, rat.INFINITY);
665 	if (isNaN(a)) return rat.copy(out, rat.INFINULL);
666 	if (a%1===0) return rat.fromInteger_copy(out, a);
667 	if ((1/a)%1===0) return rat.fromIntegerInverse_copy(out, parseInt(1/a));
668 	
669 	out[0] = 1;
670 	out[1] = 1;
671 	
672 	var neg = a < 0;
673 	if (neg) a = Math.abs(a);
674 	
675 	var m = [1, 0, 0, 1];
676 	var test = a;
677 	
678 	// traverse the Stern-Brocot tree until a match is found
679 	// this is comparing the numerator to the denominator multiplied by the target decimal
680 	var c = RAT_MAX_LOOPS;
681 	while ( out[0] !== test && c-- ) {
682 		if (out[0] > test) {
683 			m[0] += m[1];
684 			m[2] += m[3];
685 		}
686 		else {
687 			m[1] += m[0];
688 			m[3] += m[2];
689 		}
690 		out[0] = m[0] + m[1];
691 		out[1] = m[2] + m[3];
692 		test = a * out[1];
693 	}
694 	
695 	if (neg) rat.neg(out, out);
696 	return out;
697 };

711 /**
712  * Parametric sine: 2a / (1 + a²)
713  *
714  * @param {rat} out the receiving number
715  * @param {rat} a number for which to calculate the parametric sine
716  * @returns {rat} out
717  */
718 rat.sin = function(out, a) {
719 	if (a[1] === 0) return rat.copy(out, rat.ZERO);
720 	rat.scalar_multiply(out, a, 2);
721 	var d = rat.create();
722 	rat.pow(d, a, 2);
723 	rat.add(d, d, rat.ONE);
724 	rat.divide(out, out, d);
725 	return out;
726 };

728 /**
729  * Parametric cosine: (1 - a²) / (1 + a²)
730  *
731  * @param {rat} out the receiving number
732  * @param {rat} a number for which to calculate the parametric cosine
733  * @returns {rat} out
734  */
735 rat.cos = function(out, a) {
736 	if (a[1] === 0) return rat.neg(out, rat.ONE);
737 	var a2 = rat.create();
738 	rat.pow(a2, a, 2);
739 	rat.sub(out, rat.ONE, a2);
740 	var d = rat.create();
741 	rat.add(d, rat.ONE, a2);
742 	rat.divide(out, out, d);
743 	return out;
744 };

746 /**
747  * Parametric tangent: sin(a) / cos(a)

749  * @param {rat} out the receiving number
750  * @param {rat} a number for which to calculate the parametric tangent
751  * @returns {rat} out
752  */
753 rat.tan = function(out, a) {
754 	rat.scalar_multiply(out, a, 2); // out = a * 2
755 	var t = rat.create();
756 	rat.pow(t, a, 2); // t = a * a
757 	rat.scalar_multiply(t, t, 2); // t *= 2
758 	rat.add(out, out, t); // out += t
759 	rat.pow(t, a, 4); // t = a * a * a * a
760 	rat.sub(t, rat.ONE, t); // t = 1 - t
761 	rat.divide(out, out, t);
762 	return out;
763 };

765 /**
766  * Returns an Egyptian representation
767  * The returned string can be evaluated in "calc - arbitrary precision calculator"
768  *
769  * @param {rat} a number to represent as an Egyptian fraction
770  * @returns {String} string representing the most simple sum of fractions having a numerator of one, in "calc" format
771  */
772 rat.toEgyptian = function (a) {
773 	var t = rat.clone(a);
774 	rat.abs(t, t);
775 	var b = rat.floor(t);
776 	if (b) rat.sub(t, t, rat.fromInteger(b));
777 	if (!t[0]) return b.toString();
778 	if (!b) b = '';
779 	var d = 1;
780 	var f = rat.create();
781 	while (t[0] !== 1) {
782 		d++;
783 		f = rat.fromValues(1, d);
784 		if (rat.isGreaterThan(t, f)) {
785 			if (b) b += ' + ';
786 			b += rat.str(f);
787 			rat.sub(t, t, f);
788 		}
789 	}
790 	if (!t) {
791 		if (!b) return '0';
792 		return b;
793 	}
794 	return b ? b + ' + ' + rat.str(t) : rat.str(t);
795 };

797 /**
798  * Returns a Babylonian representation (base 60)
799  * The returned string can be evaluated in "calc - arbitrary precision calculator"
800  *
801  * @param {rat} a number to represent as a Babylonian fraction
802  * @returns {String} string containing the decimal representations of the base 60 digits and their powers, in "calc" format
803  */
804 rat.toBabylonian = function (a) {
805 	var s = '';
806 	 // there must be a better way to do this, avoiding conversion to decimal
807 	var t = rat.toDecimal(a);
808 	var n = parseInt(t);
809 	var r = t - n;
810 	var d = 0;
811 	var p = 0;
812 	while (n > 0) {
813 		d = n % 60;
814 		if (d) s = d + ' * 60^' + p + ( s ? ' + ' : '' ) + s;
815 		n = (n - d) / 60;
816 		p++;
817 	}
818 	p = -1;
819 	while (r > 0) {
820 		r *= 60;
821 		d = parseInt(r + .0000000000001);
822 		r = r - d;
823 		if (r < -.0000000000001) continue;
824 		if (d) s += ( s ? ' + ' : '' ) + d + ' * 60^' + p;
825 		p--;
826 	}
827 	return s ? s : '0';
828 };

830 /**
831  * Returns a string of L's and R's representing the Stern Brocot path
832  *
833  * @param {rat} a number to trace in the Stern Brocot tree
834  * @returns {String} Stern Brocot path
835  */
836 rat.traceSternBrocot = function (a) {
837 	var path = '';
838 	if (a===rat.ZERO||a===rat.ONE||a===rat.INFINITY||a===rat.INFINULL) return path;
839 
840 	var neg = rat.isNegative(a);
841 	if (neg) a[0] = -a[0];
842 	
843 	var r = rat.clone(rat.ONE);
844 	var m = [1, 0, 0, 1];
845 	
846 	var r_streak = 0;
847 	var l_streak = 0;
848 	
849 	var c = RAT_MAX_LOOPS;
850 	while ( !rat.equals(a, r) && c-- ) {
851 		if (rat.isLessThan(a, r)) {
852 			m[0] += m[1];
853 			m[2] += m[3];
854 			l_streak++;
855 			if (r_streak) {
856 				path += 'R';
857 				if (r_streak!==1) path += r_streak;
858 				r_streak = 0;
859 				path += ' ';
860 			}
861 		}
862 		else {
863 			m[1] += m[0];
864 			m[3] += m[2];
865 			r_streak++;
866 			if (l_streak) {
867 				path += 'L';
868 				if (l_streak!==1) path += l_streak;
869 				l_streak = 0;
870 				path += ' ';
871 			}
872 		}
873 		r[0] = m[0] + m[1];
874 		r[1] = m[2] + m[3];
875 	}
876 	if (l_streak) {
877 		path += 'L';
878 		if (l_streak!==1) path += l_streak;
879 	}
880 	else if (r_streak) {
881 		path += 'R';
882 		if (r_streak!==1) path += r_streak;
883 	}
884 	
885 	if (c<0) path += '...';
886 	
887 	if (neg) a[0] = -a[0];
888 	
889 	return path;
890 };
