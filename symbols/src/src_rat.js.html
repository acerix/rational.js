311 /**
312  * Normalize a rat
313  *
314  * @param {rat} out the receiving number
315  * @param {rat} a number to normalize
316  * @returns {rat} out
317  */
318 rat.normalize = function(out, a) {
319 	out[0] = a[0];
320 	out[1] = a[1];
321 	if (out[1] === 0) return out;
322 	if (out[1] < 0) {
323 		out[0] = -out[0];
324 		out[1] = -out[1];
325 	}
326 	var gcd = integer.greatest_common_divisor(Math.abs(out[0]), out[1]);
327 	if (gcd > 1) {
328 		out[0] /= gcd;
329 		out[1] /= gcd;
330 	}
331 	return out;
332 };

459 /**
460  * Returns a rat from a decimal number
461  *
462  * @param {Number} a decimal number
463  * @returns {rat} out
464  */
465 rat.fromDecimal = function (a) {
466 	
467 	a = parseFloat(a);
468 	if (a===0) return rat.clone(RAT_ZERO);
469 	if (a===1) return rat.clone(RAT_ONE);
470 	if (a===Infinity) return rat.clone(RAT_INFINITY);
471 	if (isNaN(a)) return rat.clone(RAT_INFINULL);
472 	if (a%1===0) return rat.fromInteger(a);
473 	
474 	var out = new RAT_ARRAY_TYPE(2);
475 	out[0] = 1;
476 	out[1] = 1;
477 	
478 	var neg = a < 0;
479 	if (neg) a = Math.abs(a);
480 	
481 	var m = [1, 0, 0, 1];
482 	var test = a;
483 	
484 	// traverse the Stern-Brocot tree until a match is found
485 	// this is comparing the numerator to the denominator multiplied by the target decimal
486 	while ( out[0] !== test ) {
487 		if (out[0] > test) {
488 			m[0] += m[1];
489 			m[2] += m[3];
490 		}
491 		else {
492 			m[1] += m[0];
493 			m[3] += m[2];
494 		}
495 		out[0] = m[0] + m[1];
496 		out[1] = m[2] + m[3];
497 		test = a * out[1];
498 	}
499 	
500 	if (neg) rat.neg(out, out);
501 	return out;
502 };

504 /**
505  * Creates a new rat from two random integers
506  *
507  * @returns {rat} a random rational number
508  */
509 rat.fromRandom = function() {
510 	var out = new RAT_ARRAY_TYPE(2);
511 	out[0] = Math.pow( 2, 31 ) - Math.floor( Math.random() * Math.pow( 2, 32 ) + 1 );
512 	out[1] = Math.floor( Math.random() * Math.pow( 2, 31 ) + 1 );
513 	return rat.normalize(out, out);
514 };

516 /**
517  * Parametric sine: 2a / (1 + a)
518  *
519  * @param {rat} out the receiving number
520  * @param {rat} a number for which to calculate the parametric sine
521  * @returns {rat} out
522  */
523 rat.sin = function(out, a) {
524 	if (a[1] === 0) return rat.copy(out, RAT_ZERO);
525 	rat.scalar_multiply(out, a, 2);
526 	var d = rat.create();
527 	rat.pow(d, a, 2);
528 	rat.add(d, d, RAT_ONE);
529 	rat.divide(out, out, d);
530 	return out;
531 };

533 /**
534  * Parametric cosine: (1 - a) / (1 + a)
535  *
536  * @param {rat} out the receiving number
537  * @param {rat} a number for which to calculate the parametric cosine
538  * @returns {rat} out
539  */
540 rat.cos = function(out, a) {
541 	if (a[1] === 0) return rat.neg(out, RAT_ONE);
542 	var a2 = rat.create();
543 	rat.pow(a2, a, 2);
544 	rat.sub(out, RAT_ONE, a2);
545 	var d = rat.create();
546 	rat.add(d, RAT_ONE, a2);
547 	rat.divide(out, out, d);
548 	return out;
549 };

551 /**
552  * Parametric tangent: sin(a) / cos(a)
553  *
554  * @param {rat} out the receiving number
555  * @param {rat} a number for which to calculate the parametric tangent
556  * @returns {rat} out
557  */
558 rat.tan = function(out, a) {
559 	rat.scalar_multiply(out, a, 2); // out = a * 2
560 	var t = rat.create();
561 	rat.pow(t, a, 2); // t = a * a
562 	rat.scalar_multiply(t, t, 2); // t *= 2
563 	rat.add(out, out, t); // out += t
564 	rat.pow(t, a, 4); // t = a * a * a * a
565 	rat.sub(t, RAT_ONE, t); // t = 1 - t
566 	rat.divide(out, out, t);
567 	return out;
568 };

570 /**
571  * Returns an Egyptian representation
572  * The returned string can be evaluated in "calc - arbitrary precision calculator"
573  *
574  * @param {rat} a number to represent as an Egyptian fraction
575  * @returns {String} string representing the most simple sum of fractions having a numerator of one, in "calc" format
576  */
577 rat.toEgyptian = function (a) {
578 	var t = rat.clone(a);
579 	rat.abs(t, t);
580 	var b = rat.floor(t);
581 	if (b) rat.sub(t, t, rat.fromInteger(b));
582 	if (!t[0]) return b.toString();
583 	if (!b) b = '';
584 	var d = 1;
585 	var f = rat.create();
586 	while (t[0] !== 1) {
587 		d++;
588 		f = rat.fromValues(1, d);
589 		if (rat.isGreaterThan(t, f)) {
590 			if (b) b += ' + ';
591 			b += rat.str(f);
592 			rat.sub(t, t, f);
593 		}
594 	}
595 	if (!t) {
596 		if (!b) return '0';
597 		return b;
598 	}
599 	return b ? b + ' + ' + rat.str(t) : rat.str(t);
600 };

602 /**
603  * Returns a Babylonian representation (base 60)
604  * The returned string can be evaluated in "calc - arbitrary precision calculator"
605  *
606  * @param {rat} a number to represent as a Babylonian fraction
607  * @returns {String} string containing the decimal representations of the base 60 digits and their powers, in "calc" format
608  */
609 rat.toBabylonian = function (a) {
610 	var s = '';
611 	 // there must be a better way to do this, avoiding conversion to decimal
612 	var t = rat.toDecimal(a);
613 	var n = parseInt(t);
614 	var r = t - n;
615 	var d = 0;
616 	var p = 0;
617 	while (n > 0) {
618 		d = n % 60;
619 		if (d) s = d + ' * 60^' + p + ( s ? ' + ' : '' ) + s;
620 		n = (n - d) / 60;
621 		p++;
622 	}
623 	p = -1;
624 	while (r > 0) {
625 		r *= 60;
626 		d = parseInt(r + .0000000000001);
627 		r = r - d;
628 		if (r < -.0000000000001) continue;
629 		if (d) s += ( s ? ' + ' : '' ) + d + ' * 60^' + p;
630 		p--;
631 	}
632 	return s ? s : '0';
633 };

635 /**
636  * Returns a string with the fraction in various formats
637  *
638  * @param {rat} a number to dump
639  * @returns {String} string various conversions
640  */
641 rat.dump = function(r) {
642 	var t = rat.create();
643 	return rat.str(r)
644 	+ '\ntoBabylonian\t ~ '+rat.toBabylonian(r)
645 	//+ '\ntoEgyptian\t = '+rat.toEgyptian(r)  // can be very slow
646 	+ '\ndecimal:\t ~ '+rat.toDecimal(r)
647 	+ '\nsin:\t = '+rat.sin(t, r)
648 	+ '\ncos:\t = '+rat.cos(t, r)
649 	+ '\ntan:\t = '+rat.tan(t, r)
650 	+ '\n';
651 };
