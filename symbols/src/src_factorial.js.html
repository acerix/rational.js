 37 /**
 38  * Return the factorial of n as a Javascript number
 39  *
 40  * @param {Integer} n
 41  * @returns {Integer} factorial of n
 42  */
 43 factorial.of = function(n) {
 44 	switch (typeof factorial.PREPARED[n]) {
 45 		case 'number': return factorial.PREPARED[n];
 46 		case 'string': return parseInt(factorial.PREPARED[n]);
 47 	}
 48 }
