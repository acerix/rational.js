 21 /**
 22  * @class Chemistry
 23  * @name chemistry
 24  */
 25 var chemistry = {};

 27 /**
 28  * Returns the quantum numbers of the nth electron, starting at n = 0
 29  *
 30  * @param {Integer} n the number of electrons before this one
 31  * @returns {Array} [n, l, m_l, m_s]
 32  */
 33 chemistry.quantum_numbers_of_atoms_electron = function(n) {
 34 	switch (n) {
 35 		
 36 		// 1s
 37 		case 0:		return [1, 0, +0, +1/2];
 38 		case 1:		return [1, 0, +0, -1/2]; // He
 39 		
 40 		// 2s
 41 		case 2:		return [2, 0, +0, +1/2];
 42 		case 3:		return [2, 0, +0, -1/2];
 43 		
 44 		// 2p
 45 		case 4:		return [2, 1, +0, +1/2];
 46 		case 5:		return [2, 1, +1, +1/2];
 47 		case 6:		return [2, 1, -1, +1/2];
 48 		case 7:		return [2, 1, +0, -1/2];
 49 		case 8:		return [2, 1, +1, -1/2];
 50 		case 9:		return [2, 1, -1, -1/2]; // Ne
 51 		
 52 		// 3s
 53 		case 10:	return [3, 0, +0, +1/2];
 54 		case 11:	return [3, 0, +0, -1/2];
 55 		
 56 		// 3p
 57 		case 12:	return [3, 1, +0, +1/2];
 58 		case 13:	return [3, 1, +1, +1/2];
 59 		case 14:	return [3, 1, -1, +1/2];
 60 		case 15:	return [3, 1, +0, -1/2];
 61 		case 16:	return [3, 1, +1, -1/2];
 62 		case 17:	return [3, 1, -1, -1/2]; // Ar
 63 		
 64 		// 3d
 65 		case 18:	return [3, 2, +0, +1/2];
 66 		case 19:	return [3, 2, +1, +1/2];
 67 		case 20:	return [3, 2, -1, +1/2];
 68 		case 21:	return [3, 2, +2, +1/2];
 69 		case 22:	return [3, 2, -2, +1/2];
 70 		case 23:	return [3, 2, +0, -1/2];
 71 		case 24:	return [3, 2, +1, -1/2];
 72 		case 25:	return [3, 2, -1, -1/2];
 73 		case 26:	return [3, 2, +2, -1/2];
 74 		case 27:	return [3, 2, -2, -1/2];
 75 		
 76 		// 4s
 77 		case 28:	return [4, 0, +0, +1/2];
 78 		case 29:	return [4, 0, +0, -1/2];
 79 		
 80 		// 4p
 81 		case 30:	return [4, 1, +0, +1/2];
 82 		case 31:	return [4, 1, +1, +1/2];
 83 		case 32:	return [4, 1, -1, +1/2];
 84 		case 33:	return [4, 1, +0, -1/2];
 85 		case 34:	return [4, 1, +1, -1/2];
 86 		case 35:	return [4, 1, -1, -1/2]; // Kr
 87 		
 88 		// 5s
 89 		case 36:	return [5, 0, +0, +1/2];
 90 		case 37:	return [5, 0, +0, -1/2];
 91 		
 92 		// 4d
 93 		case 38:	return [4, 2, +0, +1/2];
 94 		case 39:	return [4, 2, +1, +1/2];
 95 		case 40:	return [4, 2, -1, +1/2];
 96 		case 41:	return [4, 2, +2, +1/2];
 97 		case 42:	return [4, 2, -2, +1/2];
 98 		case 43:	return [4, 2, +0, -1/2];
 99 		case 44:	return [4, 2, +1, -1/2];
100 		case 45:	return [4, 2, -1, -1/2];
101 		case 46:	return [4, 2, +2, -1/2];
102 		case 47:	return [4, 2, -2, -1/2];
103 		
104 		// 5p
105 		case 48:	return [5, 1, +0, +1/2];
106 		case 49:	return [5, 1, +1, +1/2];
107 		case 50:	return [5, 1, -1, +1/2];
108 		case 51:	return [5, 1, +0, -1/2];
109 		case 52:	return [5, 1, +1, -1/2];
110 		case 53:	return [5, 1, -1, -1/2]; // Xe
111 		
112 		// 4f
113 		case 54:	return [4, 3, +0, +1/2];
114 		case 55:	return [4, 3, +1, +1/2];
115 		case 56:	return [4, 3, -1, +1/2];
116 		case 57:	return [4, 3, +2, +1/2];
117 		case 58:	return [4, 3, -2, +1/2];
118 		case 59:	return [4, 3, +3, +1/2];
119 		case 60:	return [4, 3, -3, +1/2];
120 		case 61:	return [4, 3, +0, -1/2];
121 		case 62:	return [4, 3, +1, -1/2];
122 		case 63:	return [4, 3, -1, -1/2];
123 		case 64:	return [4, 3, +2, -1/2];
124 		case 65:	return [4, 3, -2, -1/2];
125 		case 66:	return [4, 3, +3, -1/2];
126 		case 67:	return [4, 3, -3, -1/2];
127 				
128 		// 5d
129 		case 68:	return [5, 2, +0, +1/2];
130 		case 69:	return [5, 2, +1, +1/2];
131 		case 70:	return [5, 2, -1, +1/2];
132 		case 71:	return [5, 2, +2, +1/2];
133 		case 72:	return [5, 2, -2, +1/2];
134 		case 73:	return [5, 2, +0, -1/2];
135 		case 74:	return [5, 2, +1, -1/2];
136 		case 75:	return [5, 2, -1, -1/2];
137 		case 76:	return [5, 2, +2, -1/2];
138 		case 77:	return [5, 2, -2, -1/2];
139 		
140 		// 6s
141 		case 78:	return [6, 0, +0, +1/2];
142 		case 79:	return [6, 0, +0, -1/2];
143 		
144 		// 6p
145 		case 80:	return [6, 1, +0, +1/2];
146 		case 81:	return [6, 1, +1, +1/2];
147 		case 82:	return [6, 1, -1, +1/2];
148 		case 83:	return [6, 1, +0, -1/2];
149 		case 84:	return [6, 1, +1, -1/2];
150 		case 85:	return [6, 1, -1, -1/2]; // Rn
151 		
152 		// 5f
153 		case 86:	return [5, 3, +0, +1/2];
154 		case 87:	return [5, 3, +1, +1/2];
155 		case 88:	return [5, 3, -1, +1/2];
156 		case 89:	return [5, 3, +2, +1/2];
157 		case 90:	return [5, 3, -2, +1/2];
158 		case 91:	return [5, 3, +3, +1/2];
159 		case 92:	return [5, 3, -3, +1/2];
160 		case 93:	return [5, 3, +0, -1/2];
161 		case 94:	return [5, 3, +1, -1/2];
162 		case 95:	return [5, 3, -1, -1/2];
163 		case 96:	return [5, 3, +2, -1/2];
164 		case 97:	return [5, 3, -2, -1/2];
165 		case 98:	return [5, 3, +3, -1/2];
166 		case 99:	return [5, 3, -3, -1/2];
167 		
168 		// 6d
169 		case 100:	return [6, 2, +0, +1/2];
170 		case 101:	return [6, 2, +1, +1/2];
171 		case 102:	return [6, 2, -1, +1/2];
172 		case 103:	return [6, 2, +2, +1/2];
173 		case 104:	return [6, 2, -2, +1/2];
174 		case 105:	return [6, 2, +0, -1/2];
175 		case 106:	return [6, 2, +1, -1/2];
176 		case 107:	return [6, 2, -1, -1/2];
177 		case 108:	return [6, 2, +2, -1/2];
178 		case 109:	return [6, 2, -2, -1/2];
179 		
180 		// 7s
181 		case 110:	return [7, 0, +0, +1/2];
182 		case 111:	return [7, 0, +0, -1/2];
183 		
184 		// 7p
185 		case 112:	return [7, 1, +0, +1/2];
186 		case 113:	return [7, 1, +1, +1/2];
187 		case 114:	return [7, 1, -1, +1/2];
188 		case 115:	return [7, 1, +0, -1/2];
189 		case 116:	return [7, 1, +1, -1/2];
190 		case 117:	return [7, 1, -1, -1/2]; // Uuo
191 	
192 	}
193 }
194 
195 if(typeof(exports) !== 'undefined') {
196 	exports.chemistry = chemistry;
