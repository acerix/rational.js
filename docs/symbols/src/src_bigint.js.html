 21 /**
 22  * @class Integer of Arbitrary Size
 23  * @name bigint
 24  */
 25 var bigint = {};
 26 
 27 /**
 28  * Creates a new, empty bigint
 29  *
 30  * @returns {bigint} a new bigint
 31  */
 32 bigint.create = function() {
 33 	return [0];
 34 };
 35 
 36 /**
 37  * Creates a new bigint initialized with values from an existing number
 38  *
 39  * @param {bigint} a number to clone
 40  * @returns {bigint} a new bigintional number
 41  */
 42 bigint.clone = function(a) {
 43 	var out = [];
 44 	for (var i=0, l=a.length; i<l; i++)
 45 		out[i] = a[i];
 46 	return out;
 47 };
 48 
 49 /**
 50  * Copy the values from one bigint to another
 51  *
 52  * @param {bigint} out the receiving number
 53  * @param {bigint} a the source number
 54  * @returns {bigint} out
 55  */
 56 bigint.copy = function(out, a) {
 57 	out = [];
 58 	for (var i=0, l=a.length; i<l; i++)
 59 		out[i] = a[i];
 60 	return out;
 61 };
 62 
 63 /**
 64  * Absolute value of a bigint
 65  *
 66  * @param {bigint} out the receiving number
 67  * @param {bigint} a number to take the absolute value of
 68  * @returns {bigint} out
 69  */
 70 bigint.abs = function(out, a) {
 71 	out[0] = Math.abs(a[0]);
 72 	return out;
 73 };
 74 
 75 /**
 76  * Returns a string representation
 77  *
 78  * @param {bigint} a number to represent as a string
 79  * @returns {String} string representation of the number
 80  */
 81 bigint.str = function (a) {
 82 	return a[0].toString();
 83 };
 84 
 85 /**
 86  * Returns an integer to it's native type
 87  *
 88  * @param {bigint} a number to return as an integer
 89  * @returns {Integer} an integer or Infinity when the max size is exceeded
 90  */
 91 bigint.toInteger = function (a) {
 92 	return out[0];
 93 };
 94 
 95 /**
 96  * Returns a bigint from an integer
 97  *
 98  * @param {bigint} out the receiving number
 99  * @param {Integer} integer
100  * @returns {bigint} out
101  */
102 bigint.fromInteger = function (a) {
103 	return [parseInt(a)];
104 };
105 
106 /**
107  * Zero, the additive identity
108  *
109  * @property bigint_ZERO
110  * @type bigint_ARRAY_TYPE
111  * @static
112  * @final
113  */
114 var bigint_ZERO = bigint.fromInteger(0);
115 
116 /**
117  * One, the multiplicative identity
118  *
119  * @property bigint_ONE
120  * @type bigint_ARRAY_TYPE
121  * @static
122  * @final
123  */
124 var bigint_ONE = bigint.fromInteger(1);
